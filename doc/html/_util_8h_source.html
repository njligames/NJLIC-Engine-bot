106 inline hsv rgb2hsv(rgb in)
107 {
108  hsv out;
109  double min, max, delta;
110 
111  min = in.r < in.g ? in.r : in.g;
112  min = min < in.b ? min : in.b;
113 
114  max = in.r > in.g ? in.r : in.g;
115  max = max > in.b ? max : in.b;
116 
117  out.v = max; // v
118  delta = max - min;
119  if (delta < 0.00001)
120  {
121  out.s = 0;
122  out.h = 0; // undefined, maybe nan?
123  return out;
124  }
125  if (max > 0.0)
126  { // NOTE: if Max is == 0, this divide would cause a crash
127  out.s = (delta / max); // s
128  }
129  else
130  {
131  // if max is 0, then r = g = b = 0
132  // s = 0, v is undefined
133  out.s = 0.0;
134  out.h = NAN; // its now undefined
135  return out;
136  }
137  if (in.r >= max) // > is bogus, just keeps compilor happy
138  out.h = (in.g - in.b) / delta; // between yellow & magenta
139  else if (in.g >= max)
140  out.h = 2.0 + (in.b - in.r) / delta; // between cyan & yellow
141  else
142  out.h = 4.0 + (in.r - in.g) / delta; // between magenta & cyan
143 
144  out.h *= 60.0; // degrees
145 
146  if (out.h < 0.0)
147  out.h += 360.0;
148 
149  return out;
